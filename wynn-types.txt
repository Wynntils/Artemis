BASIC:
"accessoryType" - typ av gear
"category" -- armor/weapon/accessories
"displayName" -- replace name if item been renamed???
"identified" -- if "true" then ID numbers are fixed, not baseline
"name" -- the name unless overridden by displayName
"sockets" -- fixed, number of powder slots, an integer
"tier" -- rarity level
"type" -- gear type, string, "Helmet" etc

META:
"addedLore" - lore text
"allowCraftsman" -- bryr vi oss?
"armorColor" -- ska vara RGB. är ALLTID "160,101,64" om det finns? nonsens.
"armorType" -- vilken typ av material, guld diamant etc
"dropType" -- string "never", "NORMAL" or "lootchest"
"material" -- för allt utom armor, t.ex. "359:5"
"restrictions" -- "req", can be "Untradable", "untradable" or "Quest Item"
"skin" -- player head base64 data for some helmets
